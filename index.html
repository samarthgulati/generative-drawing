// Maple

// https://editor.p5js.org/samarthgulati/sketches/HM0-gXUJ2
var size = 400;
var center = size * 0.5;
function setup() {
  createCanvas(size, size);
}

function draw() {
  background(220, 0);
  translate(center, center);
  var a = frameCount * 0.1;
  rotate(a);
  var E = Math.E;
  translate(-50 * (sin(a) + pow(sin(2.5 * a), 3)), 0);
  point(0, 0);
}
						
					
							
// Butterfly 
// https://en.wikipedia.org/wiki/Butterfly_curve_(transcendental)

// https://editor.p5js.org/samarthgulati/sketches/wQ21I9VQx
var size = 400;
var center = size * 0.5;
function setup() {
  createCanvas(size, size);
}

function draw() {
  background(220, 1);
  translate(center, center);
  var a = frameCount * 0.005;
  rotate(a);
  var E = Math.E;
  translate(-50 * (pow(E, sin(a)) - 2 * cos(4 * a) + pow(sin((2 * a - PI) / 24), 5)), 0);
  point(0, 0);
}
